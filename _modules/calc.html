def hypot(a, b):
    """Calculate hypotenuse from the lengths of the shorter 2 sides.

    :param a: First length.
    :type a:  int or float
    :param b: Second length.
    :type b: int or float
    :return: Length of the hypotenuse.
    :rtype: int or float.
    """
    sa = squared(a)
    sb = squared(b)
    sumAB = addition(sa, sb)

    return sqroot(sumAB)
